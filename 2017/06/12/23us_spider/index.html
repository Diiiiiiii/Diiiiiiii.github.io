from bs4 import BeautifulSoup
import requests
import os,pymysql
import re

def spdier_fun(url):
reObj = requests.get(url)
html = reObj.text.encode(‘iso-8859-1’).decode(‘gbk’, errors=’ignore’)
bsObj = BeautifulSoup(html, ‘html.parser’)
return bsObj

def url_list_fun(url):
‘’’
传入小说目录url
返回各个小说章节url_list
‘’’

url_list = []
bsObj = spdier_fun(url)
for  data in bsObj.findAll('td'):
    if (data.a) != None:
        url_list += [data.a.get('href')]

return (url_list)

#print(Booktitle)
Bookcontent = bsObj.findAll('dd',id = 'contents')
for content in Bookcontent:
    print(content.get_text() + '\n')
    with open(os.getcwd()+ "\{0}.txt".format(bookname),'a',encoding='gbk',errors='ignore') as f:
        f.write(Booktitle + '\n\n')
        f.write('\t'+content.get_text() +'\n\n')
